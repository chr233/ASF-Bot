## 支持多 IPC 管理的 ASF 机器人

## 配置文件

```json
{
  "System": {
    "Debug": false,
    "Statistic": true
  },

  "Database": {
    "Generate": true,
    "LogSQL": false,

    //使用 Sqlite
    "DbType": "sqlite",
    "DbName": "data"

    //使用 Mysql, 也支持 Pgsql
    //"DbType": "mysql",
    //"DbHost": "localhost",
    //"DbPort": 3306,
    //"DbName": "db_name",
    //"DbUser": "root",
    //"DbPassword": "root"
  },

  "Telegram": {
    "Proxy": null,
    "BotToken": "123456:abcdefg",
    "AdminUsers": [],
    "DropPendingUpdates": false
  },

  "AsfIpcs": [
    {
      "Name": "example",
      "BaseUrl": "http://127.0.0.1:1242",
      "IpcPassword": "",
      "Proxy": null
    }
  ]
}
```
